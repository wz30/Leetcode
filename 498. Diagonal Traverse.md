## 498. Diagonal Traverse
- 二维数组 + 指针
```java
class Solution {
    public int[] findDiagonalOrder(int[][] matrix) {
        /*
        given [i][j]
        go NE dir: [i-1][j+1]
        go SW dir: [i+1][j-1]
        when to swap from NE to SW: hit the boundary
            1. [i][j+1](if j is valid)/[i+1][j](j is not valid)
        .... swap from SW to NE
            [i+1][j](if i is valid)/[i][j+1](if i is not valid)
        if neight i or j is valid, ends
        */
        int M = matrix.length;
        if(M==0) return new int[0];
        int N = matrix[0].length;
        if(M==0 && N==0) return new int[0];
        if(N==0) return new int[0];
        //corner
        int[] res = new int[M*N];
        int i = 0, j = 0;
        int index = 0;
        int dir = 0; //0:NE 1: SW
        while(true){
            res[index] = matrix[i][j];
            index++;
            if(dir==0){
                if(valid(i-1,j+1,M, N)){
                    i = i-1; j = j+1;
                }else{
                    dir = 1;
                    if(valid(i, j+1, M, N)){
                        j = j+1;
                    }else if(valid(i+1,j,M, N)){
                        i = i+1;
                    }else{
                        break;
                    }
                }
            }else{
                if(valid(i+1,j-1,M, N)){
                    i = i+1; j = j-1;
                }else{
                    dir = 0;
                    if(valid(i+1, j, M, N)){
                        i = i+1;
                    }else if(valid(i,j+1,M, N)){
                        j = j+1;
                    }else{
                        break;
                    }
                }
            }
        }
        
        return res;
    }
    boolean valid(int i , int j, int M, int N){
        if(i>=0 && i<M && j>=0 && j < N) return true;
        else return false;
    }
}
```
