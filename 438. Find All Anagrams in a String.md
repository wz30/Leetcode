class Solution {
    int[] t2 = new int[26];
    int[] t1 = new int[26];
    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> res = new ArrayList<>();
        
        int n1 = s.length();
        int n2 = p.length();
        for(int i = 0; i<p.length(); i++)
        {
            t2[p.charAt(i)-'a']++;
        }
        
        for(int i = 0; i<n1; i++)
        {
            if(i>=n2) t1[s.charAt(i-n2)-'a']--;
            t1[s.charAt(i)-'a']++;
            if(Arrays.equals(t1, t2))
                res.add(i-n2+1);
            
            
        }
        return res;
    }
    
}

class Solution {
    public List<Integer> findAnagrams(String s, String p) {
        //time: len(s)*len(p)
        int[] parr = new int[26];
        for(char c : p.toCharArray()){
            parr[c-'a']++;
        }
        int[] temp = new int[26];
        int i = 0, j = 0;
        List<Integer> res = new ArrayList<>();
        while(j<s.length()){
            temp[s.charAt(j)-'a']++;
            
            j++;
            if(j-i>p.length()){
               temp[s.charAt(i)-'a']--;
                i = i+1; 
            }
            if(Arrays.equals(temp,parr)){
                res.add(i);
                temp[s.charAt(i)-'a']--;
                i = i+1;
            }
        }
        
        return res;
    }
}
