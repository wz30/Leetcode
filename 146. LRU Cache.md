```java
class LRUCache {
   class Node {
    int key;
    int value;
    
    Node(int key, int value) {
        this.key = key;
        this.value = value;
        }
    }
    Map<Integer, Node> map = new HashMap<>();
    LinkedList<Node> list = new LinkedList<>();
    int capacity;
    int size;
    
    public LRUCache(int capacity) {
        this.capacity = capacity;
        this.size = 0;
    }
    
    public int get(int key) {
        Node p = map.get(key);
        if(p!=null){
            list.remove(p);
            list.addLast(p);
            return p.value;
        }
        return -1;
    }
    
    public void put(int key, int value) {
        //full, in -> remove the node(map, list), add (map, list)
        //full, not in -> remove the leaset recent node(list, map) add
        //not full, in -> remove node and add
        //not full, not in -> add
        if(size==capacity && !map.containsKey(key)){
            Node p = list.removeFirst();
            map.remove(p.key);
            
        }else if(map.containsKey(key)){
            list.remove(map.get(key));
        }else{
            size++;
        }
        Node newNode = new Node(key, value);
        list.addLast(newNode);
        map.put(key, newNode);
    }
}
 ```

 ```java
 class LRUCache {

    //we also need to maintain the stack info
    class Node{
        int key;
        int value;
        Node(int key, int value){
            this.key = key;
            this.value = value;
        }
    }
    HashMap<Integer, Node> map;
    LinkedList<Node> list;
    int capacity;
    public LRUCache(int capacity) {
        list = new LinkedList<>();
        map = new HashMap<>();
        this.capacity = capacity;
    }
    
    public int get(int key) {
        Node p = map.get(key);
        if(p!=null){
            list.remove(p);
            list.addFirst(p);
            return p.value;
        }
        return -1;
    }
    
    public void put(int key, int value) {
        if(capacity==0) return;
        Node old = map.get(key);
        if(old!=null){
            old.value = value;
            list.remove(old);
            list.addFirst(old);
            return;
        }
        if(map.size()==capacity){
            Node p = list.removeLast();
            map.remove(new Integer(p.key));
            
        }
        Node p = new Node(key, value);
        list.addFirst(p);
        map.put(key, p);
        //System.out.println(map.size());
    }
}
 ```

    - Dlinkedlist: directly remove that since the node is already the reference.

```java
class LRUCache {
    //add last, remove first, remove
    class DLinkedList{
        Node head, tail;
        DLinkedList(){
            head = null;
            tail = null;
        }
        boolean isEmpty(){
            return head==null;
        }
        void addLast(Node node){
            node.next = node.prev = null;
            //there is no dummy in head or tail
            if(isEmpty()){
                head = node;
                tail = node;
            }else{
                tail.next = node;
                node.prev = tail;
                //move tail
                tail = tail.next;
            }
        }
        //remove here we have the reference to the node
        void remove(Node node){
            if(node==null) return;
            
            if(node==head){
                removeFirst();
            }else if(node==tail){
                removeLast();
            }else{
                Node prev = node.prev;
                node.next.prev = prev;
                prev.next = node.next;
                node.prev =null;
                node.next = null;
                
            }
            
        }
        Node removeFirst(){
            if(isEmpty()) return null;
            Node node = head;
            head = head.next;
            if(head==null) tail = null;
            else{
                head.prev = null;
            }
            return node;
        }
        Node removeLast(){
            if(isEmpty()) return null;
            Node node = tail;
            tail = tail.prev;
            if(tail==null) head = null;
            else{
                tail.next = null;
            }
            return node;
        }
        
        
    }
    
   class Node {
        int key;
        int value;
        Node next, prev;
        Node(int key, int value) {
            this.key = key;
            this.value = value;
            next = null;
            prev = null;
        }
    }
    Map<Integer, Node> map = new HashMap<>();
    DLinkedList list = new DLinkedList();
    int capacity;
    int size;
    
    public LRUCache(int capacity) {
        this.capacity = capacity;
        this.size = 0;
    }
    
    public int get(int key) {
        Node p = map.get(key);
        if(p!=null){
            list.remove(p);
            list.addLast(p);
            return p.value;
        }
        return -1;
    }
    
    public void put(int key, int value) {
        //full, in -> remove the node(map, list), add (map, list)
        //full, not in -> remove the leaset recent node(list, map) add
        //not full, in -> remove node and add
        //not full, not in -> add
        if(size==capacity && !map.containsKey(key)){
            Node p = list.removeFirst();
            map.remove(p.key);
            
        }else if(map.containsKey(key)){
            list.remove(map.get(key));
        }else{
            size++;
        }
        Node newNode = new Node(key, value);
        list.addLast(newNode);
        map.put(key, newNode);
    }
}

 * obj.put(key,value);
 */
```
